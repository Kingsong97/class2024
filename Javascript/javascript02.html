05. 배열 : 데이터 불러오기 : for()

{
                    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];

                    for (let i = 0; i < arr.length; i++) {     // 초기값, 조건식, 증감값
                        console.log(arr[i]); // 실행문
                    };
}
